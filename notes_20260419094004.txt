<script>
let current=0;

/* Slide */
function next(){
  current++;
  document.getElementById("slider").style.transform="translateX(-"+(current*100)+"vw)";
  if(current==3) typeLetter();
}

/* Unlock */
function unlock(){
  if(document.getElementById("pass").value==="2007"){
    next();
    stars();
    fireworks();
  } else {
    document.getElementById("err").innerText="Wrong password ❌";
  }
}

/* Stars */
function stars(){
  let c=document.getElementById("stars");
  for(let i=0;i<30;i++){
    let s=document.createElement("div");
    s.className="star";
    s.style.top=Math.random()*100+"vh";
    s.style.left=Math.random()*100+"vw";
    c.appendChild(s);
  }
}

/* Fireworks */
function fireworks(){
  for(let i=0;i<8;i++){
    setTimeout(()=>{
      let f=document.createElement("div");
      f.className="fire";
      f.innerHTML="✨";
      f.style.left=Math.random()*100+"vw";
      f.style.top=Math.random()*100+"vh";
      document.body.appendChild(f);
      setTimeout(()=>f.remove(),800);
    },i*150);
  }
}

/* Typing */
let text="You are very special  Stay happy always ✨";
let i=0;

function typeLetter(){
  let box=document.getElementById("letter");
  let int=setInterval(()=>{
    box.innerHTML+=text[i];
    i++;
    if(i>=text.length) clearInterval(int);
  },40);
}

/* Music */
function playMusic(){
  document.getElementById("music").play();
}
